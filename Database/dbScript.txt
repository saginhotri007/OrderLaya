/****** Object:  View [dbo].[vw_RestaurantOrders]    Script Date: 13-09-2025 14:39:52 ******/

GO
SET QUOTED_IDENTIFIER ON
GO

CREATE   VIEW [dbo].[vw_RestaurantOrders]
AS
SELECT 
    o.OrderID,
    o.UserID,
    u.Name AS CustomerName,
    u.Phone AS CustomerPhone,
    o.RestaurantID,
    o.TotalAmount,
    o.Status,
    o.PaymentStatus,
    o.CreatedAt,
    a.AddressLine1 +' '+ a.AddressLine2+' '+a.Landmark+''+a.City  AS Address,
    (
        SELECT 
            oi.ItemID,
            m.Name,
            oi.Quantity,
            oi.Price
        FROM OrderItems oi
        INNER JOIN MenuItems m ON oi.ItemID = m.ItemID
        WHERE oi.OrderID = o.OrderID
        FOR JSON PATH
    ) AS Items
FROM Orders o
INNER JOIN Users u ON o.UserID = u.UserID
LEFT JOIN UserAddresses a ON o.UserID = a.UserID
